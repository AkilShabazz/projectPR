15 # Time course of the state during the pulse for the *inactivation* states
16 # and before and after the pulse for the *activation* states.
17 #
18 # The value of the state \f$v\f$ is computed according to the following
19 # equation:
20 #
21 # \f{equation}{
22 # v(t) = v_0\exp[-\beta(t-t_0)]
23 # \f}
24 # where \f$t_0\f$ is the time at which the pulse changed
25 # the value (on to off or off to on) and \f$v_0\f$ is value
26 # of the state at that time.
27 #
28 def compValOn(v0, alpha, beta, t, t0):
29  return v0 * math.exp(beta * (t0 - t))

31 ##
32 #
33 # Time course of the state during the pulse for the *activation* states
34 # and before and after the pulse for the *inactivation* states.
35 #
36 # The value of the state \f$v\f$ is computed according to the following
37 # equation:
38 #
39 # \f{equation}{
40 # v(t) = 1 + (v_0 - 1)\exp[-\alpha(t-t_0)]
41 # \f}
42 # where \f$t_0\f$ is the time at which the pulse changed
43 # the value (on to off or off to on) and \f$v_0\f$ is value
44 # of the state at that time.
45 #
46 def compValOff(v0, alpha, beta, t, t0):
47  return 1.0 + (v0 - 1.0) * math.exp(alpha * (t0 - t))
