13 # Computes the synaptic conductance
14 #
15 # - Input:
16 # + **Gmax**: the sum of individual conductances of all synapses in
17 # the compartment, in \f$\mu\f$S.
18 #
19 # + **Ron**: sum of the fraction of postsynaptic receptors
20 # that are bound to neurotransmitters of all the individual synapses
21 # that have neurotransmitters being released (during the pulse).
22 #
23 # + **Roff**: sum of the fraction of postsynaptic receptors
24 # that are bound to neurotransmitters of all the individual synapses
25 # that do not have neurotransmitters being released (before and after
26 # the pulse).
27 #
28 # - Output:
29 # + the synaptic conductance of all synapses in the compartment,
30 # in \f$\mu\f$S.
31 #
32 # It is computed by the following formula:
33 #
34 # \f{equation}{
35 # G = G_{max}(R_{on} + R_{off})
36 # \f}
37 # where \f$G\f$ is the synaptic conductance of all synapses in the compartment.
38 #
39 def compSynapCond(Gmax, Ron, Roff):
40  return Gmax * (Ron + Roff)

43 #
44 # Computes the fraction of postsynaptic receptors
45 # that are bound to neurotransmitters of all the individual synapses
46 # that have neurotransmitters being released (during the pulse).
47 #
48 # - Inputs:
49 # + **Non**: sum of the fractions of the individual conductances that are
50 # receiving neurotransmitter (during pulse) relative to
51 # the \f$G_{max}\f$ (\f$N_{on}=\limits\sum_{i=1}g_{i_{on}}/G_{max}\f$).
52 #
53 # + **rInf**: the fraction of postsynaptic receptors that
54 # would be bound to neurotransmitters after an infinite
55 # amount of time with neurotransmitter being released.
56 #
57 # + **Ron**: sum of the fraction of postsynaptic receptors
58 # that are bound to neurotransmitters of all the individual synapses
59 # that have neurotransmitters being released (during the pulse).
60 #
61 # + **t0**: instant that the last spike arrived to the compartment.
62 #
63 # + **t**: current instant, in ms.
64 #
65 # + **tauOn**: Time constant during a pulse, in ms.
66 # \f$\tau_{on}=\frac{1}{\alpha.T_{max} +\beta}\f$.
67 # - Outputs:
68 # + The fraction of postsynaptic receptors
69 # that are bound to neurotransmitters of all the individual synapses
70 # that have neurotransmitters being released
71 #
72 # It is computed by the following equation:
73 #
74 # \f{equation}{
75 # R_{on_{newValue}} = N_{on}r_{\infty}\Bigg[1-\exp\left(-\frac{t-t_0}{\tau_{on}}\right)\Bigg] + R_{on_{oldValue}}\exp\left(-\frac{t-t_0}{\tau_{on}}\right)
76 # \f}
77 #
78 def compRon(Non, rInf, Ron, t0, t, tauOn):
79  return Non * rInf + (Ron - Non * rInf) * math.exp((t0 - t) / tauOn)

82 #
83 # Computes the fraction of postsynaptic receptors
84 # that are bound to neurotransmitters of all the individual synapses
85 # that do not have neurotransmitters being released (before and after
86 # the pulse).
87 #
88 # - Inputs:
89 # + **Roff**: sum of the fraction of postsynaptic receptors
90 # that are bound to neurotransmitters of all the individual synapses
91 # that do not have neurotransmitters being released (before and after
92 # the pulse).
93 #
94 # + **t0**: instant that the last spike arrived to the compartment.
95 #
96 # + **t**: current instant, in ms.
97 #
98 # + **tauOff**: time constant after a pulse, in ms.
99 #
100 # + Output:
101 # + The fraction of postsynaptic receptors
102 # that are bound to neurotransmitters of all the individual synapses
103 # that do not have neurotransmitters being released.
104 #
105 # It is computed by the following formula:
106 #
107 # \f{equation}{
108 # R_{off_{newValue}} = R_{off_{oldValue}}\exp\left(-\frac{t - t0}{\tau_{off}} \right)
109 # \f}
110 #
111 #
112 def compRoff(Roff, t0, t, tauOff):
113  return Roff * math.exp((t0 - t) / tauOff)

116 #
117 # Computes the fraction of bound postsynaptic receptors
118 # to neurotransmitters in individual synapses when the
119 # neurotransmitter begin (begin of the pulse).
120 #
121 # - Inputs:
122 # + **ri**: the fraction of postsynaptic receptors that
123 # were bound to neurotransmitters at the last state change.
124 #
125 # + **t**: current instant, in ms.
126 #
127 # + **ti**: The instant that the last pulse began.
128 #
129 # + **tPeak**: The duration of the pulse.
130 #
131 # + **tauOff**: Time constant after a pulse, in ms.
132 #
133 # - Output:
134 # + individual synapse state value.
135 #
136 # It is computed by the following equation:
137 #
138 # \f{equation}{
139 # r_{i_{newValue}} = r_{i_{oldValue}} \exp\left(\frac{t_i+T_{dur}-t}{\tau_{off}}\right)
140 # \f}
141 #
142 def compRiStart(ri, t, ti, tPeak, tauOff):
143  return ri * math.exp((ti + tPeak - t) / tauOff)

146 #
147 # Computes the fraction of bound postsynaptic receptors
148 # to neurotransmitters in individual synapses when the
149 # neurotransmitter release stops (the pulse ends).
150 #
151 # - Inputs:
152 # + **rInf**: the fraction of postsynaptic receptors that
153 # would be bound to neurotransmitters after an infinite
154 # amount of time with neurotransmitter being released.
155 #
156 # + **ri**: the fraction of postsynaptic receptors
157 # that were bound to neurotransmitters at the last
158 # state change.
159 #
160 # + **expFinish**: Is the value of the exponential at the
161 # end of the pulse (\f$\exp(T_{dur}/\tau_{on})\f$). It is
162 # is computed before for computational efficiency.
163 #
164 # - Output:
165 # + individual synapse state value.
166 #
167 # It is computed by the following equation:
168 #
169 # \f{equation}{
170 # r_{i_{newValue}} = r_{\infty} + (r_{i_{oldValue}} - r_{\infty}) \exp\left(\frac{T_{dur}}{\tau_{on}}\right)
171 # \f}
172 #
173 def compRiStop(rInf, ri, expFinish):
174  return rInf + (ri - rInf) * expFinish
