21 # Calculates the coupling conductance between two compartments.
22 #
23 #
24 # - Inputs:
25 # + **cytR**: Cytoplasmatic resistivity in \f$\Omega\f$.cm.
26 #
27 # + **lComp1, lComp2**: length of the compartments in \f$\mu\f$m.
28 #
29 # + **dComp1, dComp2**: diameter of the compartments in \f$\mu\f$m.
30 #
31 # - Output:
32 # + coupling conductance in MS.
33 #
34 # The coupling conductance between compartment 1 and 2 is
35 # computed by the following equation:
36 #
37 # \f{equation}{
38 # g_c = \frac{2.10^2}{\frac{R_{cyt}l_1}{\pi r_1^2}+\frac{R_{cyt}l_2}{\pi r_2^2}}
39 # \f}
40 # where \f$g_c\f$ is the coupling conductance [MS], \f$R_{cyt}\f$ is the
41 # cytoplasmatic resistivity [\f$\Omega\f$.cm], \f$l_1\f$ and \f$l_2\f$
42 # are the lengths [\f$\mu\f$m] of compartments 1 and 2, respectively and
43 # \f$r_1\f$ and \f$r_2\f$ are the radius [\f$\mu\f$m] of compartments 1 and
44 # 2, respectively.
45 #
46 def calcGCoupling(cytR, lComp1, lComp2, dComp1, dComp2):
47  rAxis1 = (cytR * lComp1) / (math.pi * math.pow(dComp1/2, 2))
48  rAxis2 = (cytR * lComp2) / (math.pi * math.pow(dComp2/2, 2))
49 
50  return (1e2 * 2) / (rAxis1 + rAxis2)

57 # Computes the Coupling Matrix to be used in the dVdt function of the N compartments of the motor unit.
58 # The Matrix uses the values obtained with the function calcGcoupling.
59 #
60 # - Inputs:
61 # + **gc**: the vector with N elements, with the coupling conductance of each compartment of the Motor Unit.
62 #
63 # - Output:
64 # + the GC matrix
65 #
66 #
67 # \f{equation}{
68 # GC = \left[\begin{array}{cccccccc}
69 # -g_c[0]&g_c[0]&0&...&...&0&0&0\\
70 # g_c[0]&-g_c[0]-g_c[1]&g_c[1]&0&...&...&0&0\\
71 # \vdots&&\ddots&&...&&0&0 \\
72 # 0&...&g_c[i-1]&-g_c[i-1]-g_c[i]&g_c[i]&0&...&0\\
73 # 0&0&0&...&...&&&0\\
74 # 0&&...&&g_c[N-2]&-g_c[N-2]-g_c[N-1]&g_c[N-1]&0\\
75 # 0&...&0&&&0&g_c[N-1]&-g_c[N-1]\end{array}\right]
76 # \f}
77 #
78 def compGCouplingMatrix(gc):
79 
80  GC = np.zeros((len(gc),len(gc)))
81 
82  for i in xrange(0, len(gc)):
83  if i == 0:
84  GC[i,i:i+2] = [-gc[i], gc[i]]
85  elif i == len(gc) - 1:
86  GC[i,i-1:i+1] = [gc[i-1], -gc[i-1]]
87  else:
88  GC[i,i-1:i+2] = [gc[i-1], -gc[i-1]-gc[i], gc[i]]
89 
90 
91  return GC

95 # Function to implement the fourth order Runge-Kutta Method to solve numerically a
96 # differential equation.
97 #
98 # - Inputs:
99 # + **derivativeFunction**: function that corresponds to the derivative of the differential equation.
100 #
101 # + **t**: current instant.
102 #
103 # + **x**: current state value.
104 #
105 # + **timeStep**: time step of the solution of the differential equation, in the same unit of t.
106 #
107 # + **timeStepByTwo**: timeStep divided by two, for computational efficiency.
108 #
109 # + **timeStepBySix**: timeStep divided by six, for computational efficiency.
110 #
111 # This method is intended to solve the following differential equation:
112 #
113 # \f{equation}{
114 # \frac{dx(t)}{dt} = f(t, x(t))
115 # \f}
116 # First, four derivatives are computed:
117 #
118 # \f{align}{
119 # k_1 &= f(t,x(t))\\
120 # k_2 &= f(t+\frac{\Delta t}{2}, x(t) + \frac{\Delta t}{2}.k_1)\\
121 # k_3 &= f(t+\frac{\Delta t}{2}, x(t) + \frac{\Delta t}{2}.k_2)\\
122 # k_4 &= f(t+\Delta t, x(t) + \Delta t.k_3)
123 # \f}
124 # where \f$\Delta t\f$ is the time step of the numerical solution of the
125 # differential equation.
126 #
127 # Then the value of \f$x(t+\Delta t)\f$ is computed with:
128 #
129 # \f{equation}{
130 # x(t+\Delta t) = x(t) + \frac{\Delta t}{6}(k_1 + 2k_2 + 2k_3+k_4)
131 # \f}
132 #
133 def runge_kutta(derivativeFunction, t, x, timeStep, timeStepByTwo, timeStepBySix):
134  k1 = derivativeFunction(t, x)
135  k2 = derivativeFunction(t + timeStepByTwo, x + timeStepByTwo * k1)
136  k3 = derivativeFunction(t + timeStepByTwo, x + timeStepByTwo * k2)
137  k4 = derivativeFunction(t + timeStep, x + timeStep * k3)
138 
139  return x + timeStepBySix * (k1 + k2 + k2 + k3 + k3 + k4)
